def eucledian_distance(x, y):
    return np.sqrt(np.sum(np.power(x-y, 2)))

def eqs(x, y, err):
    return abs(y-x) <= err

# intervals, used for clustering, getting points in clusters, number of bins in the histogram
def create_intervals(filter_range, num_intervals, percent_overlap):
    intervals = []
    s,e = filter_range
    j = s
    
    interval_length = abs(e-s) / (num_intervals - (num_intervals - 1) * percent_overlap)
    
    interval_length = interval_length
    step = interval_length * (1 - percent_overlap)
    
    for i in range(num_intervals):
        j = s + (i) * step
        intervals.append([j, j + interval_length])
    intervals[len(intervals)-1][1] = e
    
    return intervals


expected1 = [(0,1), (0.333, 1.33), (0.66, 1.66), (1,2)]

# clustering, Algorithm described in mapper paper

# Single-LInkage Clustering

class Edge:
    
    def __init__(self, point1, point2, distance):
        self.A = point1
        self.B = point2
        self.distance = distance
        
    def __lt__(self,that):
        return self.distance < that.distance
    
    def __eq__(self,that):
        return self.distance == that.distance
    
    def __gt__(self,that):
        return self.distance > that.distance
    
    
def parent(parents, i):
    
    while parents[i] != i:
        i = parents[i]
        
    return i


def get_last_merge(values, num_intervals):
    
    max_edge = max(values)
    min_edge = min(values)
    
    if eqs(max_edge, min_edge, error):
        return len(values) - 1
    
    int_length = (max_edge - min_edge) / num_intervals
    bins = []
    
    for i in range(num_intervals):
        s = min_edge + (i)*int_length
        bins.append(s)

    bins.append(max_edge + 1)
    
    placements = np.digitize(values, bins)
    
    for i, b in enumerate(placements[:-1]):
        if placements[i + 1] > b + 1:
            break
    else:
        i = len(placements) - 1
            
    return i


def build_pairwise_distance_heap(data):
    
    h = []
    l = len(data)
    for i in range(l):
        for j in range(i+1, l):
            dist = eucledian_distance(data[i,], data[j,])
            heapq.heappush(h, Edge(i,j,dist))
    return h


def get_linkage(data, heap_func):
    
    edges = []
    parents = [x for x in range(len(data))]
    
    h = heap_func(data)
    
    num_clusters = len(parents)
    while num_clusters > 1:
        dist = heapq.heappop(h)
        x, y = parent(parents, dist.A), parent(parents, dist.B)
        if x != y:
            parents[x] = y
            num_clusters -= 1
            edges.append(dist)
            
    return edges


def find_clusters(num_data_points, edges):
    
    clusters, parents = [], []
    for x in range(num_data_points):
        parents.append(x)
        clusters.append(x)
        
    for e in edges:
        x, y = parent(parents, e.A), parent(parents, e.B)
        parents[x] = y
    
    for i, c in enumerate(clusters):
        clusters[i] = parent(parents, c)
    
    return clusters


def slic(data, heap_func, num_intervals):
    
    if len(data) == 0:
        return []
    if len(data) == 1:
        return [0]
    
    edges = get_linkage(data, heap_func)
    
    edge_lengths = [e.distance for e in edges]    
    last_merge = get_last_merge(edge_lengths, num_intervals)
    edges = edges[:last_merge + 1]
    clusters = find_clusters(len(data), edges)
    return clusters

def test_get_last_merge():

    values1 = [1, 2, 3, 4, 5]

    assert get_last_merge(values1, num_intervals1) == 4

def test_build_pairwise_distance_heap():
    
    example = np.array([[ 1.0,  1.0],
                       [2.0,  2.0],
                       [3.0,  3.0]])
    expected = [Edge(0, 1, 1.414214),
                Edge(1, 2, 1.414214),
                Edge(0, 2, 2.828427)]
    i = 0
    h = build_pairwise_distance_heap(example)
    for i in range(len(expected)):
        r = heapq.heappop(h)
        e = expected[i]
        assert e.A == r.A
        assert e.B == r.B

from collections import defaultdict

def in_range(i, preimages, intervals, filter_vals):
    for j, interval in enumerate(intervals):
        if interval[0] <= filter_vals[i] <= interval[1]:
            preimages[j].append(i)
        
def get_nodesets(data, heap_func, filter_func, clustering, 
           num_intervals, percent_overlap):
    
    filter_vals = filter_func(data)
    fmin = min(filter_vals)
    fmax = max(filter_vals)
    intervals = create_intervals((fmin, fmax), num_intervals, percent_overlap)
    preimages = {x:[] for x in range(len(intervals))}
    
    # get the indecies for each interval
    for i in range(len(data)):
        in_range(i, preimages, intervals, filter_vals)
    
    nodesets = []
    
    for i, x in enumerate(preimages):
        clusters = slic(data[preimages[x]], heap_func, num_intervals)
        clusters_dict = defaultdict(set)
        for j, n in enumerate(clusters):
            clusters_dict[str(i)+str(n)].add(preimages[x][j])
        nodesets.append(clusters_dict)
    
    return nodesets


def mapper(data, heap_func, filter_func, clustering, 
           num_intervals, percent_overlap):
    
    nodesets = get_nodesets(data, heap_func, filter_func, clustering, 
               num_intervals, percent_overlap)

    G=nx.Graph()

    for i, nodeset in enumerate(nodesets):
        for node in nodeset:
            G.add_node(node)
            for nodeset2 in nodesets[i+1:]:
                for node2 in nodeset2:
                    if len(nodeset[node].intersection(nodeset2[node2])) > 0:
                        G.add_edge(node, node2)
                        
    return G

        for i in range(N):
            d = cdist(data[(i,),:], data, **metricpar)
